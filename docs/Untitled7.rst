    RAA = cl.load_dataset('RAA')
    CL = cl.Chainladder(RAA)
    RAA_mack = cl.MackChainladder(RAA)
    summary = RAA_mack.summary()
    resid_df = RAA_mack.chainladder.get_residuals()  
    
    grid = gridplot()
    p = figure(plot_width=450, plot_height=275,title='RAA Ultimate Triangle', x_axis_label = 'Development Period', y_axis_label='Value')
    
    for num, row in enumerate(RAA.iterrows()):
        p.line([x+1 for x in range(len(RAA))],np.array(row[1].values), line_width=4, color=Spectral10[num], line_cap='round', line_join='round')
        p.line([x+1 for x in range(len(RAA)+1)],np.array(CL.full_triangle.iloc[num].values), legend=str(int(row[0])),line_dash = 'dashed', line_width=4,line_cap='round', color=Spectral10[num], line_alpha=.5)
    
    p2 = figure(plot_width=450, plot_height=275,title='RAA Triangle', x_range=p.x_range, y_range=p.y_range, x_axis_label = 'Development Period', y_axis_label='Value')
    
    for num, row in enumerate(RAA.iterrows()):
        p2.line([x+1 for x in range(len(RAA))],np.array(row[1].values), line_width=4, legend=str(int(row[0])), color=Spectral10[num], line_cap='round', line_join='round')
    
    bar = figure(plot_width=450, plot_height=300,title='Mack Summary')
    bar.vbar(x=RAA_mack.summary().index, width=0.5, bottom=0, top=RAA_mack.summary()['Latest'], color=Spectral10[1],legend='Latest')
    bar.vbar(x=RAA_mack.summary().index, width=0.5, bottom=RAA_mack.summary()['Latest'], top=RAA_mack.summary()['Latest'] + RAA_mack.summary()['IBNR'], color=Spectral10[2],legend='IBNR')
    bar.multi_line([(item,item) for item in summary.index.values.astype(int)]
    , list(zip(summary['Ultimate']+summary['Mack S.E.'], summary['Ultimate']-summary['Mack S.E.'])), color='black')
    
    resid1 = figure(plot_width=450, plot_height=275,title='Resid1', x_axis_label = 'Development Period', y_axis_label='Value')
    resid1.scatter(resid_df.index, resid_df['standard_residuals'])
    resid1.line(lowess(resid_df['standard_residuals'],resid_df.index, frac=1 if len(np.unique(resid_df.index.values))<=6 else 0.666).T[0],
                lowess(resid_df['standard_residuals'],resid_df.index, frac=1 if len(np.unique(resid_df.index.values))<=6 else 0.666).T[1],
                line_width=2, color='red', line_cap='round', line_join='round')
    
    
    grid = gridplot([p, p2, bar, resid1],  ncols=2, plot_width=450, plot_height=275)
    show(grid)
